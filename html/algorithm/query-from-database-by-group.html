3.分页结果集获取

3.1 根据原始分页查询条件计算出数据所处的位置

start = (page-1)*pageSize
end = page*pageSize

找到start,带入计算第一次小于totalArr中值的下标i，并计算start-totalArr[i]和totalArr[i]-start

找到end,带入计算第一次小于totalArr中值的下标i，并计算end-totalArr[i]和totalArr[i]-end

							/**
							 * 定位
							 * @param val 分页查询（limit m,n) 中的开始(m)和结束(m+n)的位置
							 * @param totalArr  如：[10,20,30,40] 每个数表示当前查询结束后取到的结果总数
							 * @return result[0] 表示落在哪个下标；result[1]表示距离开始的距离，result[2]表示距离结束的距离
							 */
							private int[] locate(int val, int[] totalArr){
								int[] result = new int[]{-1,-1,-1};
								for(int i = 0; i < totalArr.length; i ++){
									if(val <= totalArr[i]) {
										result[0] = i;
										//计算距离前后的位置
										if(i == 0){
											result[1] = val;
										}else{
											result[1] = val - totalArr[i -1];
										}
										result[2] = totalArr[i] - val;
										return result;
									}
								}
								return result;
							}

							private int[] getIndexIn2DArray(int x ,int m, int n){
